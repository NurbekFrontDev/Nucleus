17/// A Windows security descriptor
18///
19/// Since the layout of security descriptors is unstable, this is an *opaque type*, i.e. it is only meant to be used
20/// behind a reference or pointer.
21///
22/// You can configure the security descriptor of a state upon creation through the
23/// [`StateCreation::security_descriptor`](crate::manage::StateCreation::security_descriptor) method.
24///
25/// See RFC [1861-extern-types](https://rust-lang.github.io/rfcs/1861-extern-types.html) for some background on opaque
26/// types.
27///
28/// See [Security Descriptors](https://learn.microsoft.com/en-us/windows/win32/secauthz/security-descriptors) for
29/// details about security descriptors.
30#[repr(C)]
31pub struct SecurityDescriptor {
32    _opaque: [u8; 0],
33}
34
35impl SecurityDescriptor {
36    /// Returns a mutable raw pointer to the security descriptor for use in FFI
37    pub(crate) fn as_ptr(&self) -> PSECURITY_DESCRIPTOR {
38        PSECURITY_DESCRIPTOR(self as *const Self as *mut c_void)
39    }
40}
41
42impl Drop for SecurityDescriptor {
43    fn drop(&mut self) {
44        unreachable!("SecurityDescriptor is an opaque type");
45    }
46}
47
48impl Debug for SecurityDescriptor {
49    fn fmt(&self, f: &mut Formatter<'_>) -> fmt::Result {
50        // Hide the `_opaque` field
51        f.debug_struct("SecurityDescriptor").finish()
52    }
53}

55/// An owned security descriptor allocated on the local heap
56///
57/// Unlike [`Box<SecurityDescriptor>`], this allocates memory on the
58/// [local heap](https://learn.microsoft.com/en-us/windows/win32/memory/global-and-local-functions).
59///
60/// There are two ways to create a [`BoxedSecurityDescriptor`]:
61/// - via the [`BoxedSecurityDescriptor::create_everyone_generic_all`] method
62/// - via the [`FromStr`] implementation of [`BoxedSecurityDescriptor`]
63#[derive(Debug)]
64pub struct BoxedSecurityDescriptor {
65    ptr: NonNull<SecurityDescriptor>,
66}
67
68// SAFETY:
69// - It is safe to deallocate an object on the local heap that was allocated on a different thread
70// - `BoxedSecurityDescriptor` owns a `SecurityDescriptor`, which is `Send`
71unsafe impl Send for BoxedSecurityDescriptor {}
72
73// SAFETY:
74// `BoxedSecurityDescriptor` derefs to a `SecurityDescriptor`, which is `Sync`
75unsafe impl Sync for BoxedSecurityDescriptor {}
76
77impl BoxedSecurityDescriptor {
78    /// Creates a security descriptor granting `GENERIC_ALL` access to `Everyone`
79    ///
80    /// This is the security descriptor used by default when creating states.
81    ///
82    /// The created security descriptor corresponds to the Security Descriptor String `D:(A;;GA;;;WD)`, meaning it has:
83    /// - no owner
84    /// - no group
85    /// - no System Access Control List (SACL)
86    /// - a Discretionary Access Control List (`D` = DACL) with a single Access Control Entry (ACE) granting (`A`) the
87    ///   `GENERIC_ALL` access right (`GA`) to `Everyone` (`WD` = World)
88    ///
89    /// # Errors
90    /// Returns an error if creating the security descriptor fails
91    pub fn create_everyone_generic_all() -> io::Result<Self> {
92        "D:(A;;GA;;;WD)".parse()
93    }
94}
95
96impl FromStr for BoxedSecurityDescriptor {
97    type Err = io::Error;
98
99    /// Parses a [`BoxedSecurityDescriptor`] from a Security Descriptor String
100    ///
101    /// See
102    /// [Security Descriptor String Format](https://learn.microsoft.com/en-us/windows/win32/secauthz/security-descriptor-string-format)
103    /// for details.
104    fn from_str(s: &str) -> io::Result<Self> {
105        let mut psecurity_descriptor = PSECURITY_DESCRIPTOR::default();
106        let string_security_descriptor = CWideString::new(s);
107
108        // SAFETY:
109        // - The pointer in the first argument points to a valid null-terminated wide string because it comes from a
110        //   live `CWideString`
111        // - The pointer in the third argument is valid for writes of `PSECURITY_DESCRIPTOR` because it comes from a
112        //   live mutable reference
113        // - The pointer in the fourth argument can be `NULL` according to documentation
114        unsafe {
115            ConvertStringSecurityDescriptorToSecurityDescriptorW(
116                string_security_descriptor.as_pcwstr(),
117                SDDL_REVISION,
118                &mut psecurity_descriptor,
119                None,
120            )
121        }?;
122
123        Ok(Self {
124            ptr: NonNull::new(psecurity_descriptor.0 as *mut SecurityDescriptor)
125                .expect("ConvertStringSecurityDescriptorToSecurityDescriptorW returned `NULL` security descriptor"),
126        })
127    }
128}

130impl Drop for BoxedSecurityDescriptor {
131    fn drop(&mut self) {
132        // Note: This can fail, but we have to silently ignore the error because `drop` must not fail
133
134        // SAFETY:
135        // - `self.ptr` points to a local memory object because it was returned from
136        //   `ConvertStringSecurityDescriptorToSecurityDescriptorW`
137        // - `self.ptr` has not been freed yet
138        unsafe { LocalFree(Some(HLOCAL(self.ptr.as_ptr() as *mut c_void))) };
139    }
140}
141
142impl Deref for BoxedSecurityDescriptor {
143    type Target = SecurityDescriptor;
144
145    fn deref(&self) -> &SecurityDescriptor {
146        // SAFETY:
147        // - `self.ptr` is trivially properly aligned as `mem::align_of::<SecurityDescriptor>() == 1`
148        // - `self.ptr` points to a valid `SecurityDescriptor` because it was returned from
149        //   `ConvertStringSecurityDescriptorToSecurityDescriptorW` and has not been freed yet
150        // - the `SecurityDescriptor` pointed to by `self.ptr` is live during the lifetime of the produced reference
151        //   because it is not freed before `self` is dropped
152        unsafe { self.ptr.as_ref() }
153    }
154}

159    }

164        self

168/// Borrowing security descriptors from
169/// [`windows_permissions`](https://docs.rs/windows_permissions/latest/windows_permissions)
170#[cfg(feature = "windows_permissions")]
171mod impl_windows_permissions {
172    use super::*;
173
174    impl Borrow<SecurityDescriptor> for windows_permissions::SecurityDescriptor {
175        fn borrow(&self) -> &SecurityDescriptor {
176            let ptr = self as *const windows_permissions::SecurityDescriptor as *const SecurityDescriptor;
177
178            // SAFETY:
179            // - `ptr` comes from a live reference to a `windows_permissions::SecurityDescriptor` with the same lifetime
180            //   as the produced reference
181            // - `windows_permissions::SecurityDescriptor` has the same invariants as `SecurityDescriptor`
182            unsafe { &*ptr }
183        }
184    }
185
186    impl AsRef<SecurityDescriptor> for windows_permissions::SecurityDescriptor {
187        fn as_ref(&self) -> &SecurityDescriptor {
188            self.borrow()
189        }
190    }
191
192    impl Borrow<SecurityDescriptor> for &windows_permissions::SecurityDescriptor {
193        fn borrow(&self) -> &SecurityDescriptor {
194            (*self).borrow()
195        }
196    }
197
198    impl Borrow<SecurityDescriptor> for windows_permissions::LocalBox<windows_permissions::SecurityDescriptor> {
199        fn borrow(&self) -> &SecurityDescriptor {
200            (**self).borrow()
201        }
202    }
203
204    impl AsRef<SecurityDescriptor> for windows_permissions::LocalBox<windows_permissions::SecurityDescriptor> {
205        fn as_ref(&self) -> &SecurityDescriptor {
206            self.borrow()
207        }
208    }
209}
