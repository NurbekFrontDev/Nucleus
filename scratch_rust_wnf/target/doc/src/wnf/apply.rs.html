16impl<T> OwnedState<T>
17where
18    T: Read<T> + NoUninit,
19{
20    /// Applies a transformation to the data of this state
21    ///
22    /// This essentially queries the state data, applies the given transformation closure and then updates the state
23    /// data with the transformed value. However, it tries to do so in a loop using change stamps to ensure that no
24    /// concurrent update happens between querying and updating the state data. This means that the given closure may be
25    /// called multiple times. Note that it does *not* reliably avoid concurrent updates while the actual update is
26    /// happening. If another concurrent update makes the size of the state data exceed the internal capacity of the
27    /// state (causing a reallocation), it may happen that this update does not have the desired effect on the state
28    /// data.
29    ///
30    /// The closure receives an owned `T` on the stack, requiring `T: Sized`. In order to receive a `Box<T>` for
31    /// `T: ?Sized`, use the [`apply_boxed`](OwnedState::apply_boxed) method.
32    ///
33    /// For fallible transformations, i.e. when the transformation closure returns a [`Result<D, E>`], use the
34    /// [`try_apply`](OwnedState::try_apply) method.
35    ///
36    /// The return value is the value with which the state was ultimately updated, i.e. the return value of the last
37    /// call to the given closure.
38    ///
39    /// For example, to increment the value of a state by one and return the incremented value:
40    /// ```
41    /// # fn main() -> Result<(), Box<dyn std::error::Error>> {
42    /// use std::io;
43    ///
44    /// use wnf::{AsState, OwnedState};
45    ///
46    /// fn increment<S>(state: S) -> io::Result<u32>
47    /// where
48    ///     S: AsState<Data = u32>,
49    /// {
50    ///     state.as_state().apply(|value| value + 1)
51    /// }
52    ///
53    /// let state = OwnedState::create_temporary()?;
54    /// state.set(&42)?;
55    ///
56    /// let new_data = increment(&state)?;
57    /// assert_eq!(new_data, 43);
58    /// # Ok(()) }
59    /// ```
60    ///
61    /// # Errors
62    /// Returns an error if querying or updating fails
63    pub fn apply<D, F>(&self, transform: F) -> io::Result<D>
64    where
65        D: Borrow<T>,
66        F: FnMut(T) -> D,
67    {
68        self.raw.apply(transform)
69    }
70
71    /// Applies a fallible transformation to the data of this state
72    ///
73    /// This essentially queries the state data, applies the given transformation closure and then (in case it succeeds)
74    /// updates the state data with the transformed value. However, it tries to do so in a loop using change stamps
75    /// to ensure that no concurrent update happens between querying and updating the state data. This means that the
76    /// given closure may be called multiple times. Note that it does *not* reliably avoid concurrent updates while the
77    /// actual update is happening. If another concurrent update makes the size of the state data exceed the
78    /// internal capacity of the state (causing a reallocation), it may happen that this update does not have the
79    /// desired effect on the state data.
80    ///
81    /// The closure receives an owned `T` on the stack, requiring `T: Sized`. In order to receive a `Box<T>` for
82    /// `T: ?Sized`, use the [`apply_boxed`](OwnedState::apply_boxed) method.
83    ///
84    /// For infallible transformations, i.e. when the transformation closure returns a `D` rather than a [`Result<D,
85    /// E>`], use the [`apply`](OwnedState::apply) method.
86    ///
87    /// The return value is the value with which the state was ultimately updated, i.e. the return value of the last
88    /// call to the given closure.
89    ///
90    /// For example, to increment the value of a state by one, unless a maximum is reached, and return the incremented
91    /// value:
92    /// ```
93    /// # fn main() -> Result<(), Box<dyn std::error::Error>> {
94    /// use std::io;
95    ///
96    /// use wnf::{AsState, OwnedState};
97    ///
98    /// fn try_increment<S>(state: S, max: u32) -> io::Result<u32>
99    /// where
100    ///     S: AsState<Data = u32>,
101    /// {
102    ///     state.as_state().try_apply(|value| {
103    ///         if value < max {
104    ///             Ok(value + 1)
105    ///         } else {
106    ///             Err("maximum reached")
107    ///         }
108    ///     })
109    /// }
110    ///
111    /// let state = OwnedState::create_temporary()?;
112    /// state.set(&42)?;
113    ///
114    /// let new_data = try_increment(&state, 43)?;
115    /// assert_eq!(new_data, 43);
116    ///
117    /// let result = try_increment(&state, 43);
118    /// assert!(result.is_err());
119    /// # Ok(()) }
120    /// ```
121    ///
122    /// # Errors
123    /// Returns an error if querying or updating fails, or if the transformation closure returns an error
124    pub fn try_apply<D, E, F>(&self, transform: F) -> io::Result<D>
125    where
126        D: Borrow<T>,
127        E: Into<Box<dyn Error + Send + Sync>>,
128        F: FnMut(T) -> Result<D, E>,
129    {
130        self.raw.try_apply(transform)
131    }
132}

138    /// Applies a transformation to the data of this state as a box
139    ///
140    /// This essentially queries the state data, applies the given transformation closure and then updates the state
141    /// data with the transformed value. However, it tries to do so in a loop using change stamps to ensure that no
142    /// concurrent update happens between querying and updating the state data. This means that the given closure may
143    /// be called multiple times. Note that it does *not* reliably avoid concurrent updates while the actual update is
144    /// happening. If another concurrent update makes the size of the state data exceed the internal capacity of the
145    /// state (causing a reallocation), it may happen that this update does not have the desired effect on the state
146    /// data.
147    ///
148    /// The closure receives a [`Box<T>`]. In order to receive an owned `T` on the stack (requiring `T: Sized`), use the
149    /// [`apply`](OwnedState::apply) method.
150    ///
151    /// For fallible transformations, i.e. when the transformation closure returns a [`Result<D, E>`], use the
152    /// [`try_apply_boxed`](OwnedState::try_apply_boxed) method.
153    ///
154    /// The return value is the value with which the state was ultimately updated, i.e. the return value of the last
155    /// call to the given closure.
156    ///
157    /// For example, to extend a slice by one element and return the extended slice as a [`Vec<u32>`]:
158    /// ```
159    /// # fn main() -> Result<(), Box<dyn std::error::Error>> {
160    /// use std::io;
161    ///
162    /// use wnf::{AsState, OwnedState};
163    ///
164    /// fn extend<S>(state: S) -> io::Result<Vec<u32>>
165    /// where
166    ///     S: AsState<Data = [u32]>,
167    /// {
168    ///     state.as_state().apply_boxed(|slice| {
169    ///         let mut vec = slice.into_vec();
170    ///         vec.push(42);
171    ///         vec
172    ///     })
173    /// }
174    ///
175    /// let state = OwnedState::<[u32]>::create_temporary()?;
176    /// state.set(&[1, 2, 3])?;
177    ///
178    /// let new_data = extend(&state)?;
179    /// assert_eq!(*new_data, [1, 2, 3, 42]);
180    /// # Ok(()) }
181    /// ```
182    ///
183    /// # Errors
184    /// Returns an error if querying or updating fails
185    pub fn apply_boxed<D, F>(&self, transform: F) -> io::Result<D>
186    where
187        D: Borrow<T>,
188        F: FnMut(Box<T>) -> D,
189    {
190        self.raw.apply_boxed(transform)
191    }
192
193    /// Applies a fallible transformation to the data of this state as a box
194    ///
195    /// This essentially queries the state data, applies the given transformation closure and then (in case it succeeds)
196    /// updates the state data with the transformed value. However, it tries to do so in a loop using change stamps
197    /// to ensure that no concurrent update happens between querying and updating the state data. This means that the
198    /// given closure may be called multiple times. Note that it does *not* reliably avoid concurrent updates while the
199    /// actual update is happening. If another concurrent update makes the size of the state data exceed the internal
200    /// capacity of the state (causing a reallocation), it may happen that this update does not have the desired effect
201    /// on the state data.
202    ///
203    /// The closure receives a [`Box<T>`]. In order to receive an owned `T` on the stack (requiring `T: Sized`), use the
204    /// [`try_apply`](OwnedState::try_apply) method.
205    ///
206    /// For infallible transformations, i.e. when the transformation closure returns a `D` rather than a [`Result<D,
207    /// E>`], use the [`apply_boxed`](OwnedState::apply_boxed) method.
208    ///
209    /// The return value is the value with which the state was ultimately updated, i.e. the return value of the last
210    /// call to the given closure.
211    ///
212    /// For example, to extend a slice by one element, unless a maximum length is reached, and return the extended
213    /// slice as a [`Vec<u32>`]:
214    /// ```
215    /// # fn main() -> Result<(), Box<dyn std::error::Error>> {
216    /// use std::io;
217    ///
218    /// use wnf::{AsState, OwnedState};
219    ///
220    /// fn try_extend<S>(state: S, max_len: usize) -> io::Result<Vec<u32>>
221    /// where
222    ///     S: AsState<Data = [u32]>,
223    /// {
224    ///     state.as_state().try_apply_boxed(|slice| {
225    ///         if slice.len() < max_len {
226    ///             let mut vec = slice.into_vec();
227    ///             vec.push(42);
228    ///             Ok(vec)
229    ///         } else {
230    ///             Err("maximum length reached")
231    ///         }
232    ///     })
233    /// }
234    ///
235    /// let state = OwnedState::<[u32]>::create_temporary()?;
236    /// state.set(&[1, 2, 3])?;
237    ///
238    /// let new_data = try_extend(&state, 4)?;
239    /// assert_eq!(*new_data, [1, 2, 3, 42]);
240    ///
241    /// let result = try_extend(&state, 4);
242    /// assert!(result.is_err());
243    /// # Ok(()) }
244    /// ```
245    ///
246    /// # Errors
247    /// Returns an error if querying or updating fails
248    pub fn try_apply_boxed<D, E, F>(&self, transform: F) -> io::Result<D>
249    where
250        D: Borrow<T>,
251        E: Into<Box<dyn Error + Send + Sync>>,
252        F: FnMut(Box<T>) -> Result<D, E>,
253    {
254        self.raw.try_apply_boxed(transform)
255    }
256}

362impl<T> RawState<T>
363where
364    T: ?Sized,
365{
366    /// Applies a transformation to the data of this state, passing a value of type `D` to it
367    ///
368    /// If `T: Sized`, then `D` can be either `T` or `Box<T>`.
369    /// If `T: !Sized`, then `D` must be `Box<T>`.
370    pub(crate) fn apply_as<ReadInto, WriteFrom, F>(self, mut transform: F) -> io::Result<WriteFrom>
371    where
372        WriteFrom: Borrow<T>,
373        T: Read<ReadInto> + NoUninit,
374        F: FnMut(ReadInto) -> WriteFrom,
375    {
376        self.try_apply_as(|data| Ok::<_, Infallible>(transform(data)))
377    }
378
379    /// Applies a fallible transformation to the data of this state, passing a value of type `D` to it
380    ///
381    /// If `T: Sized`, then `D` can be either `T` or `Box<T>`.
382    /// If `T: !Sized`, then `D` must be `Box<T>`.
383    fn try_apply_as<ReadInto, WriteFrom, E, F>(self, mut transform: F) -> io::Result<WriteFrom>
384    where
385        WriteFrom: Borrow<T>,
386        T: Read<ReadInto> + NoUninit,
387        E: Into<Box<dyn Error + Send + Sync>>,
388        F: FnMut(ReadInto) -> Result<WriteFrom, E>,
389    {
390        let result = loop {
391            let (data, change_stamp) = self.query_as()?.into_data_change_stamp();
392            let result = transform(data).map_err(io::Error::other)?;
393            if self.update(result.borrow(), change_stamp)? {
394                break result;
395            }
396        };
397
398        Ok(result)
399    }
