15impl<T> OwnedState<T>
16where
17    T: Read<T>,
18{
19    /// Queries the data of this state
20    ///
21    /// This produces an owned `T` on the stack and hence requires `T: Sized`. In order to produce a `Box<T>` for
22    /// `T: ?Sized`, use the [`get_boxed`](OwnedState::get_boxed) method.
23    ///
24    /// This returns the data of the state without a change stamp. In order to query both the data and the change
25    /// stamp, use the [`query`](OwnedState::query) method.
26    ///
27    /// # Errors
28    /// Returns an error if querying fails, including the case that the queried data is not a valid `T`
29    pub fn get(&self) -> io::Result<T> {
30        self.raw.get()
31    }
32
33    /// Queries the data of this state together with its change stamp
34    ///
35    /// This produces an owned `T` on the stack and hence requires `T: Sized`. In order to produce a `Box<T>` for
36    /// `T: ?Sized`, use the [`query_boxed`](OwnedState::query_boxed) method.
37    ///
38    /// This returns the data of the state together with its change stamp as a [`StampedData<T>`]. In order to
39    /// only query the data, use the [`get`](OwnedState::get) method.
40    ///
41    /// # Errors
42    /// Returns an error if querying fails, including the case that the queried data is not a valid `T`
43    pub fn query(&self) -> io::Result<StampedData<T>> {
44        self.raw.query()
45    }
46}

174impl<T> RawState<T>
175where
176    T: ?Sized,
177{
178    /// Queries the change stamp of this state
179    pub(crate) fn change_stamp(self) -> io::Result<ChangeStamp> {
180        Ok(self.cast::<OpaqueData>().query()?.change_stamp())
181    }
182
183    /// Queries the data of this state as a value of type `D`
184    ///
185    /// If `T: Sized`, then `D` can be either `T` or `Box<T>`.
186    /// If `T: !Sized`, then `D` must be `Box<T>`.
187    pub(crate) fn query_as<D>(self) -> io::Result<StampedData<D>>
188    where
189        T: Read<D>,
190    {
191        let reader = |ptr, size| {
192            let mut change_stamp = ChangeStamp::default();
193            let mut read_size = size as u32;
194
195            // SAFETY:
196            // - The pointer in the first argument points to a valid `u64` because it comes from a live reference
197            // - The pointer in the second argument is either a null pointer or points to a valid `GUID` by the
198            //   guarantees of `TypeId::as_ptr`
199            // - The pointer in the fourth argument is valid for writes of `u32` because it comes from a live mutable
200            //   reference
201            // - The pointer in the fifth argument is valid for writes of `read_size` by the precondition of `reader`
202            //   (see `Read::from_reader`) and `read_size == size`
203            // - The pointer in the sixth argument points to a valid `u32` because it comes from a live reference
204            // - The pointer in the sixth argument is valid for writes of `u32` because it comes from a live mutable
205            //   reference
206            let result = unsafe {
207                ntapi::NtQueryWnfStateData(
208                    &self.state_name.opaque_value(),
209                    self.type_id.as_ptr(),
210                    ptr::null(),
211                    change_stamp.as_mut_ptr(),
212                    ptr,
213                    &mut read_size,
214                )
215            };
216
217            if result.is_err() && (result != STATUS_BUFFER_TOO_SMALL || read_size as usize <= size) {
218                debug!(
219                     target: ntapi::TRACING_TARGET,
220                     ?result,
221                     input.state_name = %self.state_name,
222                     input.type_id = %self.type_id,
223                     "NtQueryWnfStateData",
224                );
225
226                Err(io::Error::from_raw_os_error(result.0))
227            } else {
228                // Here we know that either of the following conditions holds:
229                // a) `result.is_ok()`
230                // b) `result == STATUS_BUFFER_TOO_SMALL && read_size as usize > size`
231                debug!(
232                    target: ntapi::TRACING_TARGET,
233                    ?result,
234                    input.state_name = %self.state_name,
235                    input.type_id = %self.type_id,
236                    output.change_stamp = %change_stamp,
237                    output.buffer_size = read_size,
238                    "NtQueryWnfStateData",
239                );
240
241                Ok((read_size as usize, change_stamp))
242            }
243        };
244
245        // SAFETY:
246        // When `reader(ptr, size)` returns `Ok((read_size, _))` with `read_size <= size`,
247        // - then condition a) (see above) holds,
248        // - hence the call to `NtQueryWnfStateData` succeeded,
249        // - hence by the assumption on `NtQueryWnfStateData`, the memory range of size `read_size` starting at `ptr` is
250        //   initialized,
251        // so the safety condition of `T::from_reader` is satisfied
252        let result = unsafe { T::from_reader(reader) };
253
254        Ok(result?.into())
255    }
256}
